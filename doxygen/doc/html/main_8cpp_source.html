552  /************************************************************************/
553  bool calibrate()
554  {
555  Bottle cmd;
556  bool ok_calibration=false;
557 
558  acquire_superq();
559 
560  while (!superq_received)
561  Time::delay(0.1);
562 
563  yInfo()<<"Received superquadric: "<<superq_b.toString();
564 
565  Vector sup(11,0.0);
566  sup=getBottle(superq_b, cmd);
567 
568  if (!superq_b.isNull())
569  {
570  ok_acq=true;
571 
572  if (superq_b.size()>0 && norm(sup.subVector(0,7))>0.0)
573  {
574  if (superqOk(sup))
575  superq_ok=superq_received=true;
576  else
577  superq_ok=superq_received=false;
578  }
579  else
580  {
581  yError()<<"Zero superquadric";
582  superq_ok=superq_received=false;
583  }
584  }
585 
586  if (superq_ok)
587  {
588  yDebug()<<"Plane height: "<<sup[7];
589  }
590 
591  if (sup[7]<0.0 && sup[7]>-0.18)
592  {
593  Bottle cmd, reply_pose;
594  cmd.addString("set_options");
595 
596  Bottle &ext_content=cmd.addList();
597  Bottle &content=ext_content.addList();
598  content.addString("plane");
599  Bottle &plane_bottle=content.addList();
600  plane_bottle.addDouble(0.0); plane_bottle.addDouble(0.0); plane_bottle.addDouble(1.0); plane_bottle.addDouble(-sup[7]);
601 
602  cmd.addString("pose");
603 
604  ok_calibration=graspRpc.write(cmd, reply_pose);
605 
606  yDebug()<<"Ok set plane "<<ok_calibration;
607 
608  }
609 
610  start_from_scratch();
611 
612  return ok_calibration;
613  }
